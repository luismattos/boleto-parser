 83class Digitavel:
 84    """
 85    Representa a linha digitável de um boleto bancário.
 86    Responsável por parsing, validação e extração de campos conforme Febraban.
 87    """
 88    valor: str
 89    _campos: Optional[CamposDigitavel] = None
 90
 91    def __post_init__(self):
 92        """
 93        Normaliza a linha digitável removendo espaços e pontos
 94        """
 95        # Normalizar: remover espaços e pontos, mantendo apenas dígitos
 96        self.valor = regex.sub(r'[.\s]', '', self.valor or "")
 97        
 98        # Extrair campos se tiver tamanho adequado
 99        if len(self.valor) >= 47:
100            self._extrair_campos()
101    
102    def _extrair_campos(self):
103        """
104        Extrai os campos da linha digitável normalizada (47 dígitos)
105        Estrutura correta:
106        Campo 1: 0-9 (9 dígitos) + DV 9 (1 dígito) = 10
107        Campo 2: 10-20 (10 dígitos) + DV 20 (1 dígito) = 11
108        Campo 3: 21-31 (10 dígitos) + DV 31 (1 dígito) = 11
109        Campo 4: 32 (1 dígito)
110        Campo 5: 33-46 (14 dígitos)
111        """
112        bloco_campo1 = self.valor[0:10]   # 9 dígitos + DV
113        bloco_campo2 = self.valor[10:21]  # 10 dígitos + DV
114        bloco_campo3 = self.valor[21:32]  # 10 dígitos + DV
115        dv_geral = self.valor[32:33]      # 1 dígito
116        fator_valor_e_valor = self.valor[33:47]  # 14 dígitos
117
118        self._campos = CamposDigitavel(
119            bloco_campo1=bloco_campo1,
120            bloco_campo2=bloco_campo2,
121            bloco_campo3=bloco_campo3,
122            dv_geral=dv_geral,
123            fator_valor_e_valor=fator_valor_e_valor
124        )
125
126    def _calcular_modulo_10(self, numero: str) -> int:
127        """
128        Calcula o dígito verificador usando Módulo 10
129        
130        Algoritmo:
131        1. Multiplica cada dígito alternadamente por 2 e 1 (da direita para esquerda)
132        2. Se o resultado > 9, subtrai-se 9 (ou soma-se os dígitos do resultado)
133        3. Soma todos os resultados
134        4. DV = 10 - (soma % 10)
135        """
136        numero_invertido = numero[::-1]
137        soma = 0
138        
139        for i, digito in enumerate(numero_invertido):
140            peso = 2 if i % 2 == 0 else 1
141            resultado = int(digito) * peso
142            
143            # Se resultado > 9, subtrai-se 9
144            if resultado > 9:
145                resultado = resultado - 9
146            
147            soma += resultado
148        
149        # Calcula o DV
150        dv = 10 - (soma % 10)
151        if dv == 10:
152            dv = 0
153        
154        return dv
155
156    def _calcular_modulo_11(self, numero: str) -> int:
157        """
158        Calcula o dígito verificador usando Módulo 11
159        
160        Algoritmo:
161        1. Multiplica cada dígito por pesos de 2 a 9 (cíclicos, da direita para esquerda)
162        2. Soma todos os resultados
163        3. DV = 11 - (soma % 11)
164        4. Se resultado for 0, 10 ou 11, DV = 1
165        """
166        numero_invertido = numero[::-1]
167        soma = 0
168        
169        for i, digito in enumerate(numero_invertido):
170            peso = (i % 8) + 2  # Pesos de 2 a 9 (cíclicos)
171            resultado = int(digito) * peso
172            soma += resultado
173        
174        # Calcula o DV
175        resto = soma % 11
176        if resto == 0:
177            dv = 1
178        elif resto == 1:
179            dv = 0
180        else:
181            dv = 11 - resto
182        
183        return dv
184
185    def _validar_campo(self, campo: str) -> bool:
186        """
187        Valida o DV de um campo usando Módulo 10
188        """
189        if len(campo) < 2:
190            return False
191        
192        campo_sem_dv = campo[:-1]
193        dv_esperado = int(campo[-1])
194        dv_calculado = self._calcular_modulo_10(campo_sem_dv)
195        
196        return dv_calculado == dv_esperado
197
198    def _validar_dv_geral(self, codigo_barras: str) -> bool:
199        """
200        Valida o DV geral do código de barras usando Módulo 11
201        """
202        if len(codigo_barras) < 2:
203            return False
204        
205        codigo_sem_dv = codigo_barras[:-1]
206        dv_esperado = int(codigo_barras[-1])
207        dv_calculado = self._calcular_modulo_11(codigo_sem_dv)
208        
209        return dv_calculado == dv_esperado

211    def validar(self) -> bool:
212        """
213        Valida a linha digitável:
214        - Deve conter apenas dígitos
215        - Deve ter 47 dígitos
216        - DVs dos campos devem ser válidos
217        - DV geral deve ser válido
218        """
219        if not self.valor.isdigit():
220            return False
221        if len(self.valor) != 47:
222            return False
223        
224        if not self._campos:
225            return False
226        
227        # Validação dos campos (Módulo 10)
228        if not self._validar_campo(self._campos.bloco_campo1):
229            return False
230        if not self._validar_campo(self._campos.bloco_campo2):
231            return False
232        if not self._validar_campo(self._campos.bloco_campo3):
233            return False
234        
235        # Validação DV geral (Módulo 11)
236        codigo_barras = self._gerar_codigo_barras()
237        if not self._validar_dv_geral(codigo_barras):
238            return False
239        
240        return True

242    def _gerar_codigo_barras(self) -> str:
243        """
244        Gera o código de barras a partir da linha digitável
245        """
246        if not self._campos or len(self.valor) < 47:
247            return ""
248        
249        # Composição do código de barras (44 dígitos + DV geral)
250        banco_moeda = self.valor[0:4]
251        campo_livre = self._campos.campo_livre
252        fator_valor = self._campos.fator_valor_e_valor
253        
254        codigo_sem_dv = banco_moeda + campo_livre + fator_valor
255        dv_geral = self._calcular_modulo_11(codigo_sem_dv)
256        
257        return codigo_sem_dv + str(dv_geral)

259    def corrigir_dv(self) -> str:
260        """
261        Sugere uma linha digitável com DVs corrigidos
262        """
263        if not self._campos or len(self.valor) < 47:
264            return self.valor
265        
266        # Corrigir campo 2
267        campo2_sem_dv = self._campos.bloco_campo2[:-1]
268        dv_campo2_calculado = self._calcular_modulo_10(campo2_sem_dv)
269        campo2_corrigido = campo2_sem_dv + str(dv_campo2_calculado)
270        
271        # Corrigir campo 3
272        campo3_sem_dv = self._campos.bloco_campo3[:-1]
273        dv_campo3_calculado = self._calcular_modulo_10(campo3_sem_dv)
274        campo3_corrigido = campo3_sem_dv + str(dv_campo3_calculado)
275        
276        # Corrigir DV geral
277        banco_moeda = self.valor[0:4]
278        campo_livre = self._campos.bloco_campo1[4:9] + campo2_corrigido[:-1] + campo3_corrigido[:-1]
279        fator_valor = self._campos.fator_valor_e_valor
280        
281        codigo_sem_dv = banco_moeda + campo_livre + fator_valor
282        dv_geral_calculado = self._calcular_modulo_11(codigo_sem_dv)
283        campo4_corrigido = str(dv_geral_calculado)
284        
285        # Montar digitável corrigido
286        digitavel_corrigido = (
287            self._campos.bloco_campo1 +
288            campo2_corrigido +
289            campo3_corrigido +
290            campo4_corrigido +
291            self._campos.fator_valor_e_valor
292        )
293        
294        return digitavel_corrigido

332    def gerar_digitavel_valido(banco="033", valor=150.00, vencimento_dias=30):
333        """
334        Gera um código digitável válido seguindo as especificações Febraban
335        Campo livre é preenchido com '1' (dummy), mas todos os DVs são válidos.
336        """
337        # Data base: 07/10/1997
338        data_base = datetime(1997, 10, 7)
339        data_vencimento = datetime.now() + timedelta(days=vencimento_dias)
340        fator_vencimento = (data_vencimento - data_base).days
341        
342        # Valor em centavos (10 dígitos)
343        valor_centavos = int(valor * 100)
344        valor_str = f"{valor_centavos:010d}"
345        
346        # Fator de vencimento (4 dígitos)
347        fator_str = f"{fator_vencimento:04d}"
348        
349        # Campo livre dummy (25 dígitos)
350        campo_livre = "1" * 25
351        
352        # Montar campos para linha digitável (estrutura correta)
353        campo1_base = banco + "9" + campo_livre[:4]  # 3+1+4 = 8 dígitos
354        campo2_base = campo_livre[4:14]  # 10 dígitos
355        campo3_base = campo_livre[14:24]  # 10 dígitos
356        
357        # Calcular DVs dos campos 1, 2, 3 (Módulo 10)
358        digitavel = Digitavel("")
359        dv1 = digitavel._calcular_modulo_10(campo1_base)
360        dv2 = digitavel._calcular_modulo_10(campo2_base)
